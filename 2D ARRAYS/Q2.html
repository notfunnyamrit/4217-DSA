Enter two Matrices from user, add the matrices and display the answer


    #include 

        int main() {
            int row, col;
        
            // Input for the number of rows and columns for the matrices
            printf("Enter the number of rows: ");
            scanf("%d", &row);
            printf("Enter the number of columns: ");
            scanf("%d", &col);
        
            // Initialize two matrices of the same size
            int matrix1[row][col], matrix2[row][col], result[row][col];
        
            // Input for the elements of the first matrix
            printf("Enter the elements of the first matrix:\n");
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    scanf("%d", &matrix1[i][j]);
                }
            }
        
            // Input for the elements of the second matrix
            printf("Enter the elements of the second matrix:\n");
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    scanf("%d", &matrix2[i][j]);
                }
            }
        
            // Add the two matrices
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    result[i][j] = matrix1[i][j] + matrix2[i][j];
                }
            }
        
            // Display the result matrix
            printf("The sum of the two matrices is:\n");
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    printf("%d\t", result[i][j]);
                }
                printf("\n");
            }
        
            return 0;
        }
