            struct Node {
                int data;
                struct Node* next;
                struct Node* prev;
            };
            
            struct Node* head = NULL;
            
            void insertAtBegin(int value) {
                struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
                newNode->data = value;
                newNode->next = newNode->prev = newNode;
            
                if (head == NULL) {
                    head = newNode;
                } else {
                    newNode->next = head;
                    newNode->prev = head->prev;
                    head->prev->next = newNode;
                    head->prev = newNode;
                    head = newNode;
                }
            }
            
            void insertAtPos(int value, int pos) {
                if (pos < 0) {
                    printf("Invalid position\n");
                    return;
                }
            
                if (pos == 0) {
                    insertAtBegin(value);
                    return;
                }
            
                struct Node* temp = head;
                struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
                newNode->data = value;
            
                for (int i = 0; i < pos - 1; i++) {
                    temp = temp->next;
                    if (temp == head) {
                        printf("Invalid position\n");
                        free(newNode);
                        return;
                    }
                }
            
                newNode->next = temp->next;
                newNode->prev = temp;
                temp->next->prev = newNode;
                temp->next = newNode;
            }
            
            void insertAtEnd(int value) {
                insertAtPos(value, -1);
            }
            
            void deleteAtBegin() {
                if (head == NULL) {
                    printf("List is empty\n");
                    return;
                }
            
                struct Node* temp = head;
            
                if (head->next == head) {
                    head = NULL;
                } else {
                    head = head->next;
                    head->prev = temp->prev;
                    temp->prev->next = head;
                }
            
                free(temp);
            }
            
            void deleteAtEnd() {
                if (head == NULL) {
                    printf("List is empty\n");
                    return;
                }
            
                struct Node* temp = head->prev;
            
                if (head->next == head) {
                    head = NULL;
                } else {
                    temp->prev->next = head;
                    head->prev = temp->prev;
                }
            
                free(temp);
            }
            
            void deleteAtPos(int pos) {
                if (pos < 0 || head == NULL) {
                    printf("Invalid operation\n");
                    return;
                }
            
                if (pos == 0) {
                    deleteAtBegin();
                    return;
                }
            
                struct Node* temp = head;
            
                for (int i = 0; i < pos; i++) {
                    temp = temp->next;
                    if (temp == head) {
                        printf("Invalid position\n");
                        return;
                    }
                }
            
                temp->prev->next = temp->next;
                temp->next->prev = temp->prev;
            
                free(temp);
            }
            
            int isEmpty() {
                return (head == NULL);
            }
            
            void display() {
                if (head == NULL) {
                    printf("List is empty\n");
                    return;
                }
            
                struct Node* temp = head;
            
                do {
                    printf("%d ", temp->data);
                    temp = temp->next;
                } while (temp != head);
            
                printf("\n");
            }
            
            int search(int value) {
                if (head == NULL) {
                    return 0;
                }
            
                struct Node* temp = head;
            
                do {
                    if (temp->data == value) {
                        return 1;
                    }
                    temp = temp->next;
                } while (temp != head);
            
                return 0;
            }
            
            int main() {
                insertAtBegin(10);
                insertAtBegin(20);
                insertAtEnd(30);
                insertAtPos(5, 1);
            
                display();  // Output: 20 5 10 30
            
                deleteAtPos(2);
            
                display();  // Output: 20 10 30
            
                deleteAtBegin();
            
                display();  // Output: 10 30
            
                deleteAtEnd();
            
                display();  // Output: 10
            
                if (isEmpty()) {
                    printf("List is empty\n");
                } else {
                    printf("List is not empty\n");
                }
            
                if (search(10)) {
                    printf("Element 10 found\n");
                } else {
                    printf("Element 10 not found\n");
                }
            
                return 0;
            }
