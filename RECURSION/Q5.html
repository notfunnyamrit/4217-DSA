        int findLargestElement(int arr[], int size) {
            if (size == 1) {
                return arr[0];
            }
        
            int largestInRest = findLargestElement(arr, size - 1);
        
            return (arr[size - 1] > largestInRest) ? arr[size - 1] : largestInRest;
        }
        
        int main() {
            int size;
            printf("Enter the size of the array: ");
            scanf("%d", &size);
        
            if (size <= 0) {
                printf("Array size should be a positive integer.\n");
                return 1;
            }
        
            int array[size];
        
            printf("Enter the elements of the array:\n");
            for (int i = 0; i < size; i++) {
                scanf("%d", &array[i]);
            }
        
            int largest = findLargestElement(array, size);
            printf("The largest element in the array is: %d\n", largest);
        
            return 0;
        }
