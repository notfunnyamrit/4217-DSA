    #include 

        int sumOfDigits(int num) {
            if (num == 0) {
                return 0;
            } else {
                return (num % 10) + sumOfDigits(num / 10);
            }
        }
        
        int main() {
            int number;
        
            printf("Enter a number: ");
            scanf("%d", &number);
        
            if (number < 0) {
                printf("Please enter a non-negative number.\n");
                return 1;
            }
        
            int sum = sumOfDigits(number);
            printf("The sum of the digits of %d is: %d\n", number, sum);
        
            return 0;
        }
