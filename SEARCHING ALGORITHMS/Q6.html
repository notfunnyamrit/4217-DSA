        // Function to perform binary search on the string
        int binarySearch(char str[], char target, int left, int right) {
            if (left > right) {
                return -1; // Character not found
            }
        
            int mid = (left + right) / 2;
        
            if (str[mid] == target) {
                return mid; // Character found at index mid
            } else if (str[mid] > target) {
                return binarySearch(str, target, left, mid - 1); // Search left half
            } else {
                return binarySearch(str, target, mid + 1, right); // Search right half
            }
        }
        
        int main() {
            char str[100];
            char target;
            int length, index;
        
            printf("Enter a string: ");
            fgets(str, sizeof(str), stdin);
        
            length = strlen(str);
            if (str[length - 1] == '\n') {
                str[length - 1] = '\0'; // Removing the newline character from the string
            }
        
            printf("Enter a character to search for: ");
            scanf(" %c", &target);
        
            index = binarySearch(str, target, 0, strlen(str) - 1);
        
            if (index != -1) {
                printf("Character '%c' found at index: %d\n", target, index);
            } else {
                printf("Character '%c' not found in the string.\n", target);
            }
        
            return 0;
        }
