        int binarySearch(char str[], char target) {
            int left = 0;
            int right = strlen(str) - 1;
        
            while (left <= right) {
                int mid = (left + right) / 2;
        
                if (str[mid] == target) {
                    return mid; // Character found at index mid
                } else if (str[mid] < target) {
                    left = mid + 1; // Update left boundary
                } else {
                    right = mid - 1; // Update right boundary
                }
            }
        
            return -1; // Character not found
        }
        
        int main() {
            char str[100];
            char target;
            int index;
        
            printf("Enter a string: ");
            fgets(str, sizeof(str), stdin);
        
            printf("Enter a character to search for: ");
            scanf(" %c", &target);
        
            index = binarySearch(str, target);
        
            if (index != -1) {
                printf("Character '%c' found at index: %d\n", target, index);
            } else {
                printf("Character '%c' not found in the string.\n", target);
            }
        
            return 0;
        }
