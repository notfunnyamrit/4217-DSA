CALCULATE THE BEST AND WORST COMPLEXITY OF THE LINEAR SEARCHING ALGORITHM WITH EXACT TIME TAKEN BY THE ALGORITHM ON INTEGER ARRAY


    #include

        // Linear search function
        int linearSearch(int arr[], int size, int target) {
            for (int i = 0; i < size; i++) {
                if (arr[i] == target) {
                    return i; // Element found
                }
            }
            return -1; // Element not found
        }
        
        int main() {
            // Generate an array of random integers
            int size = 100000; // Adjust the array size as needed
            int arr[size];
            srand(time(0)); // Seed for random numbers
        
            for (int i = 0; i < size; i++) {
                arr[i] = rand() % 1000; // Filling the array with random integers (0-999)
            }
        
            int bestCaseElement = arr[0]; // Best case: Element at the start of the array
            int worstCaseElement = 9999; // Element not present in the array
        
            clock_t start, end;
            double cpu_time_used;
        
            // BEST CASE: Element at the beginning of the array
            start = clock();
            int bestCaseResult = linearSearch(arr, size, bestCaseElement);
            end = clock();
        
            cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            if (bestCaseResult != -1) {
                printf("Best Case: Element %d found at index: %d\n", bestCaseElement, bestCaseResult);
            } else {
                printf("Best Case: Element %d not found in the array.\n", bestCaseElement);
            }
        
            printf("Best Case Time taken: %f seconds\n\n", cpu_time_used);
        
            // WORST CASE: Element not present in the array
            start = clock();
            int worstCaseResult = linearSearch(arr, size, worstCaseElement);
            end = clock();
        
            cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            if (worstCaseResult != -1) {
                printf("Worst Case: Element %d found at index: %d\n", worstCaseElement, worstCaseResult);
            } else {
                printf("Worst Case: Element %d not found in the array.\n", worstCaseElement);
            }
        
            printf("Worst Case Time taken: %f seconds\n", cpu_time_used);
        
            return 0;
        }
