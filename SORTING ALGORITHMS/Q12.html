        // Function to swap two strings
        void swapStrings(char* str1, char* str2) {
            char temp[100];
            strcpy(temp, str1);
            strcpy(str1, str2);
            strcpy(str2, temp);
        }
        
        // Recursive Selection Sort on a string array
        void recursiveSelectionSort(char arr[][100], int size, int idx) {
            if (idx >= size - 1)
                return;
        
            int min_idx = idx;
            for (int j = idx + 1; j < size; j++) {
                if (strcmp(arr[j], arr[min_idx]) < 0)
                    min_idx = j;
            }
        
            swapStrings(arr[min_idx], arr[idx]);
            recursiveSelectionSort(arr, size, idx + 1);
        }
        
        int main() {
            int size;
        
            printf("Enter the number of strings in the array: ");
            scanf("%d", &size);
        
            char arr[size][100];
        
            printf("Enter %d strings:\n", size);
            for (int i = 0; i < size; i++) {
                scanf("%s", arr[i]);
            }
        
            // Calculate the time for worst case
            clock_t start = clock();
            recursiveSelectionSort(arr, size, 0);
            clock_t end = clock();
            double cpu_time_used_worst = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            // Best case: Already sorted array
            // To simulate the best case, copy the array to another array and sort
            char arr_best[size][100];
            for (int i = 0; i < size; i++) {
                strcpy(arr_best[i], arr[i]);
            }
        
            start = clock();
            recursiveSelectionSort(arr_best, size, 0);
            end = clock();
            double cpu_time_used_best = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            printf("\nSorted Array: ");
            for (int i = 0; i < size; i++) {
                printf("%s ", arr[i]);
            }
            printf("\n");
        
            printf("Time taken for worst case: %f seconds\n", cpu_time_used_worst);
            printf("Time taken for best case: %f seconds\n", cpu_time_used_best);
        
            return 0;
        }
