        // Function to swap two strings
        void swapStrings(char* str1, char* str2) {
            char temp[100];
            strcpy(temp, str1);
            strcpy(str1, str2);
            strcpy(str2, temp);
        }
        
        // Function to partition the array and return the partition index
        int partition(char arr[][100], int low, int high) {
            char pivot[100];
            strcpy(pivot, arr[high]);
            int i = low - 1;
        
            for (int j = low; j < high; j++) {
                if (strcmp(arr[j], pivot) < 0) {
                    i++;
                    swapStrings(arr[i], arr[j]);
                }
            }
            swapStrings(arr[i + 1], arr[high]);
            return i + 1;
        }
        
        // Iterative Quick Sort function for sorting a string array
        void iterativeQuickSort(char arr[][100], int size) {
            int stack[size];
            int top = -1;
            stack[++top] = 0;
            stack[++top] = size - 1;
        
            while (top >= 0) {
                int high = stack[top--];
                int low = stack[top--];
        
                int p = partition(arr, low, high);
        
                if (p - 1 > low) {
                    stack[++top] = low;
                    stack[++top] = p - 1;
                }
        
                if (p + 1 < high) {
                    stack[++top] = p + 1;
                    stack[++top] = high;
                }
            }
        }
        
        int main() {
            int size;
        
            printf("Enter the number of strings in the array: ");
            scanf("%d", &size);
        
            char arr[size][100];
        
            printf("Enter %d strings:\n", size);
            for (int i = 0; i < size; i++) {
                scanf("%s", arr[i]);
            }
        
            // Calculate the time for worst case
            clock_t start = clock();
            iterativeQuickSort(arr, size);
            clock_t end = clock();
            double cpu_time_used_worst = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            // Best case: Already sorted array
            // To simulate the best case, copy the array to another array and sort
            char arr_best[size][100];
            for (int i = 0; i < size; i++) {
                strcpy(arr_best[i], arr[i]);
            }
        
            start = clock();
            iterativeQuickSort(arr_best, size);
            end = clock();
            double cpu_time_used_best = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            printf("\nSorted Array: ");
            for (int i = 0; i < size; i++) {
                printf("%s ", arr[i]);
            }
            printf("\n");
        
            printf("Time taken for worst case: %f seconds\n", cpu_time_used_worst);
            printf("Time taken for best case: %f seconds\n", cpu_time_used_best);
        
            return 0;
