        // Function to merge two sorted subarrays into one
        void merge(int arr[], int left, int mid, int right) {
            int n1 = mid - left + 1;
            int n2 = right - mid;
        
            int Left[n1], Right[n2];
        
            // Fill the left and right subarrays with values from the main array
            for (int i = 0; i < n1; i++)
                Left[i] = arr[left + i];
            for (int j = 0; j < n2; j++)
                Right[j] = arr[mid + 1 + j];
        
            int i = 0, j = 0, k = left;
        
            // Merge the left and right subarrays into the main array in a sorted manner
            while (i < n1 && j < n2) {
                if (Left[i] <= Right[j]) {
                    arr[k] = Left[i];
                    i++;
                } else {
                    arr[k] = Right[j];
                    j++;
                }
                k++;
            }
        
            // Copy remaining elements of Left[] and Right[] into the main array
            while (i < n1) {
                arr[k] = Left[i];
                i++;
                k++;
            }
            while (j < n2) {
                arr[k] = Right[j];
                j++;
                k++;
            }
        }
        
        // Function to perform recursive merge sort
        void recursiveMergeSort(int arr[], int left, int right) {
            if (left < right) {
                // Calculate the middle point to divide the array into two halves
                int mid = left + (right - left) / 2;
        
                // Recursively sort the two halves
                recursiveMergeSort(arr, left, mid);
                recursiveMergeSort(arr, mid + 1, right);
        
                // Merge the sorted halves
                merge(arr, left, mid, right);
            }
        }
        
        int main() {
            int size;
        
            printf("Enter the size of the array: ");

            int arr[size];
        
            printf("Enter %d elements:\n", size);
            for (int i = 0; i < size; i++) {

            // Calculate the time for worst case
            clock_t start = clock();
            recursiveMergeSort(arr, 0, size - 1);
            clock_t end = clock();
            double cpu_time_used_worst = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            // Calculate the time for best case
            start = clock();
            recursiveMergeSort(arr, 0, size - 1);
            clock_t end_best = clock();
            double cpu_time_used_best = ((double)(end_best - start)) / CLOCKS_PER_SEC;
