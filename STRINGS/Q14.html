ENTER THE STRING FROM THE USER AND PRINT THE DIFFERENCE OF ASCII VALUE OF EACH CHARACTER


    #include 

        int main() {
            char inputString[1000];
        
            printf("Enter a string: ");
            fgets(inputString, sizeof(inputString), stdin);
        
            printf("ASCII differences between consecutive characters:\n");
        
            for (int i = 0; inputString[i] != '\0' && inputString[i + 1] != '\0'; i++) {
                int diff = inputString[i + 1] - inputString[i];
                printf("%d ", diff);
            }
        
            return 0;
        }
