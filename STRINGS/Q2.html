ENTER THE STRING FROM USER AND CONVERT IN LOWERCASE 


    #include 

        int main() {
            char str[100];
            int i;
        
            printf("Enter a string: ");
            scanf("%[^\n]s", str);
        
            printf("The input string is: %s\n", str);
        
            for(i = 0; str[i]; i++){
                if(str[i] >= 'A' && str[i] <= 'Z'){
                    str[i] += 32;
                }
            }
        
            printf("The string in lowercase is: %s\n", str);
        
            return 0;
        }
