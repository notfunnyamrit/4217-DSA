#include 
    #include 
    
    // Recursive function to perform linear search for a character in a string
    int linearSearch(char str[], char target, int index) {
        if (str[index] == '\0') {
            return -1; // Character not found
        }
        if (str[index] == target) {
            return index; // Character found at this index
        }
        return linearSearch(str, target, index + 1); // Recursively search the next index
    }
    
    int main() {
        char str[100];
        char target;
    
        // Input the string
        printf("Enter a string: ");
        fgets(str, sizeof(str), stdin);
        str[strcspn(str, "\n")] = '\0'; // Remove the newline character
    
        // Input the character to search for
        printf("Enter a character to search for: ");
        scanf(" %c", &target); // Note the space before %c to skip leading whitespace
    
        int index = linearSearch(str, target, 0);
    
        if (index != -1) {
            printf("Character '%c' found at index %d.\n", target, index);
        } else {
            printf("Character '%c' not found in the string.\n", target);
        }
    
        return 0;
    }
