    #include 

        // Recursive function to perform linear search for an integer in an array
        int linearSearch(int arr[], int target, int index, int size) {
            if (index == size) {
                return -1; // Element not found
            }
            if (arr[index] == target) {
                return index; // Element found at this index
            }
            return linearSearch(arr, target, index + 1, size); // Recursively search the next index
        }
        
        int main() {
            int n, target;
        
            // Input the size of the array
            printf("Enter the size of the array: ");
            scanf("%d", &n);
        
            // Input the array elements
            int arr[n];
            printf("Enter %d integers separated by spaces:\n", n);
            for (int i = 0; i < n; i++) {
                scanf("%d", &arr[i]);
            }
        
            // Input the integer value to search for
            printf("Enter the integer value to search for: ");
            scanf("%d", &target);
        
            int index = linearSearch(arr, target, 0, n);
        
            if (index != -1) {
                printf("Element %d found at index %d.\n", target, index);
            } else {
                printf("Element %d not found in the array.\n", target);
            }
        
            return 0;
        }
