        // Function to swap two elements in an array
        void swap(int *a, int *b) {
            int temp = *a;
            *a = *b;
            *b = temp;
        }
        
        // Recursive function for a single pass of Bubble Sort
        void onePassBubbleSort(int arr[], int size, int n) {
            if (n == 1) {
                return;
            }
        
            for (int i = 0; i < n - 1; i++) {
                if (arr[i] > arr[i + 1]) {
                    swap(&arr[i], &arr[i + 1]);
                }
            }
        
            onePassBubbleSort(arr, size, n - 1);
        }
        
        // Recursive Bubble Sort function
        void recursiveBubbleSort(int arr[], int size) {
            if (size == 1) {
                return;
            }
        
            onePassBubbleSort(arr, size, size);
            recursiveBubbleSort(arr, size - 1);
        }
        
        int main() {
            int size;
        
            printf("Enter the size of the array: ");
            scanf("%d", &size);
        
            int arr[size];
        
            printf("Enter %d elements:\n", size);
            for (int i = 0; i < size; i++) {
                scanf("%d", &arr[i]);
            }
        
            // Calculate the time for worst case (i.e., array is reverse sorted)
            clock_t start = clock();
            recursiveBubbleSort(arr, size);
            clock_t end = clock();
            double cpu_time_used_worst = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            // Calculate the time for best case (i.e., array is already sorted)
            start = clock();
            recursiveBubbleSort(arr, size);
            end = clock();
            double cpu_time_used_best = ((double)(end - start)) / CLOCKS_PER_SEC;
        
            printf("\nSorted Array: ");
            for (int i = 0; i < size; i++) {
                printf("%d ", arr[i]);
            }
            printf("\n");
        
            printf("Time taken for worst case: %f seconds\n", cpu_time_used_worst);
            printf("Time taken for best case: %f seconds\n", cpu_time_used_best);
        
            return 0;
        }
